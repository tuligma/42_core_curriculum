<ctype.h>:

isalpha 	- Checks if a character is an alphabet letter.
isdigit 	- Checks if a character is a decimal digit.
isalnum 	- Checks if a character is an alphanumeric character.
isascii 	- Checks if a character is a 7-bit ASCII character.
isprint 	- Checks if a character is a printable character.

<string.h>:

strlen 		- Determines the length of a string.
memset 		- Fills a block of memory with a specific value.
memcpy 		- Copies a block of memory from one location to another.
memmove 	- Moves a block of memory from one location to another, handling overlapping memory areas.
strlcpy 	- Copies a string into a destination buffer while ensuring that it does not exceed the buffer size.
strlcat 	- Concatenates a string to the end of another string while ensuring that the resulting string does not exceed a specified size.
strchr 		- Searches for the first occurrence of a character in a string.
strrchr 	- Searches for the last occurrence of a character in a string.
strncmp 	- Compares a specified number of characters from two strings.
memchr 		- Searches for the first occurrence of a character in a block of memory.
memcmp 		- Compares two blocks of memory up to a specified size.
strnstr 	- Searches for the first occurrence of a substring within a string, limited to a specified size.
strdup 		- Creates a duplicate of a string.

<stdlib.h>:

calloc 		- Allocates and initializes a block of memory.
atoi 		- Converts a string to an integer.
